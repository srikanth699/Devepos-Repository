  <script>
    const students = ["Alice Johnson", "Bob Smith", "Charlie Lee", "Diana King"];
    
    const studentTable = document.getElementById("student-table");

    students.forEach((name, index) => {
      const row = document.createElement("tr");
      row.innerHTML = `
        <td>${name}</td>
        <td class="attendance-buttons">
          <button class="present" onclick="markAttendance(${index}, 'Present')">Present</button>
          <button class="absent" onclick="markAttendance(${index}, 'Absent')">Absent</button>
        </td>
      `;
      studentTable.appendChild(row);
    });

    const attendanceData = {};

    function markAttendance(index, status) {
      const buttonsCell = studentTable.rows[index].cells[1];
      const buttons = buttonsCell.querySelectorAll("button");

      buttons.forEach(btn => btn.style.opacity = "0.4");
      buttons.forEach(btn => btn.disabled = false);

      const selectedButton = Array.from(buttons).find(btn => btn.textContent === status);
      selectedButton.style.opacity = "1";
      selectedButton.disabled = true;

      attendanceData[students[index]] = status;
    }

    function submitAttendance() {
      const date = document.getElementById("attendance-date").value;
      if (!date) {
        alert("Please select a date before submitting attendance.");
        return;
      }

      console.log("Date:", date);
      console.log("Attendance Data:", attendanceData);

      alert("Attendance submitted successfully!");
      // You can replace this with an API POST request
    }
  </script>
